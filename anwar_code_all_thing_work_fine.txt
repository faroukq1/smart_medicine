#include <Wire.h>
#include <SparkFunLSM6DS3.h> 

// Updated with your exact physical scan results
const uint8_t LSM6DS3_ADDR        = 0x6B; // Found via scan
const uint8_t MAX30205_ADDR       = 0x4C; // Found via scan
const uint8_t MAX30102_ADDR       = 0x57; // Found via scan

LSM6DS3 imuSensor(I2C_MODE, LSM6DS3_ADDR);

// Driver Utilities Declarations
void setupMax30102();
void readMax30102Data(uint32_t &red, uint32_t &ir);
float readMAX30205Temp();
void writeReg(uint8_t address, uint8_t reg, uint8_t value);

void setup() {
  Serial.begin(115200);
  while (!Serial) delay(10); 

  Serial.println("\n==================================================");
  Serial.println("   RUNNING SENSORS WITH LIVE HARDWARE ADDRESSES   ");
  Serial.println("==================================================");

  // Initialize I2C on pins 4 and 5
  Wire.begin(4, 5);

  // 1. INITIALIZE LSM6DS3 (0x6B)
  if (imuSensor.begin() == 0) {
    Serial.println("[Init] LSM6DS3: Successfully initialized at 0x6B!");
  } else {
    Serial.println("[Init] LSM6DS3: Error communicating at 0x6B");
  }

  // 2. INITIALIZE MAX30102 (0x57)
  Wire.beginTransmission(MAX30102_ADDR);
  if (Wire.endTransmission() == 0) {
    setupMax30102();
    Serial.println("[Init] MAX30102: Successfully initialized at 0x57!");
  } else {
    Serial.println("[Init] MAX30102: Error communicating at 0x57");
  }

  // 3. INITIALIZE MAX30205 (0x4C)
  Wire.beginTransmission(MAX30205_ADDR);
  if (Wire.endTransmission() == 0) {
    Serial.println("[Init] MAX30205: Successfully initialized at 0x4C!");
  } else {
    Serial.println("[Init] MAX30205: Error communicating at 0x4C");
  }

  Serial.println("\n--- Starting Real-time Sensor Data Readout (1s) ---");
}

void loop() {
  Serial.println("\n──────────────────────────────────────────────────");

  // 1. Read LSM6DS3 (Accelerometer)
  float ax = imuSensor.readFloatAccelX();
  float ay = imuSensor.readFloatAccelY();
  float az = imuSensor.readFloatAccelZ();
  Serial.printf("[LSM6DS3]  X: %+0.2fg  |  Y: %+0.2fg  |  Z: %+0.2fg\n", ax, ay, az);

  // 2. Read MAX30205 (Temperature)
  float temp = readMAX30205Temp();
  if (!isnan(temp)) {
    Serial.printf("[MAX30205] Temperature: %.2f °C\n", temp);
  } else {
    Serial.println("[MAX30205] Temperature: Reading Failed");
  }

  // 3. Read MAX30102 (Pulse Oximeter Raw Optical Values)
  uint32_t redData = 0;
  uint32_t irData = 0;
  readMax30102Data(redData, irData);
  Serial.printf("[MAX30102] Red LED Raw: %u  |  IR LED Raw: %u\n", redData, irData);
  
  if (irData < 20000) {
    Serial.println("           (Note: Place your skin/finger firmly on the MAX30102 sensor window to see values jump)");
  }

  delay(1000); 
}

  Wire.write(0x00); 
  if (Wire.endTransmission() != 0) return NAN;

  Wire.requestFrom(MAX30205_ADDR, (uint8_t)2);

void setupMax30102() {
  writeReg(MAX30102_ADDR, 0x09, 0x40); // Reset
  delay(100);
  writeReg(MAX30102_ADDR, 0x09, 0x03); // SpO2 Mode
  writeReg(MAX30102_ADDR, 0x0A, 0x27); // ADC configuration
  writeReg(MAX30102_ADDR, 0x0C, 0x24); // LED1 current
  writeReg(MAX30102_ADDR, 0x0D, 0x24); // LED2 current
  writeReg(MAX30102_ADDR, 0x04, 0x00); // Reset FIFO
  writeReg(MAX30102_ADDR, 0x05, 0x00); 
  writeReg(MAX30102_ADDR, 0x06, 0x00); 
}

void readMax30102Data(uint32_t &red, uint32_t &ir) {
  Wire.beginTransmission(MAX30102_ADDR);
  Wire.write(0x07); 
  if (Wire.endTransmission() != 0) return;

  Wire.requestFrom(MAX30102_ADDR, 6);
  if (Wire.available() == 6) {
    uint8_t r1 = Wire.read(); uint8_t r2 = Wire.read(); uint8_t r3 = Wire.read();
    uint8_t i1 = Wire.read(); uint8_t i2 = Wire.read(); uint8_t i3 = Wire.read();
    red = ((uint32_t)r1 << 16 | (uint32_t)r2 << 8 | r3) & 0x03FFFF;
    ir  = ((uint32_t)i1 << 16 | (uint32_t)i2 << 8 | i3) & 0x03FFFF;
  }
}

void writeReg(uint8_t address, uint8_t reg, uint8_t value) {
  Wire.beginTransmission(address);
  Wire.write(reg);
  Wire.write(value);
  Wire.endTransmission();
}
