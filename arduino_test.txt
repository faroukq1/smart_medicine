// ─────────────────────────────────────────────────────────────────────────────
//  USER CONFIGURATION
// ─────────────────────────────────────────────────────────────────────────────
#define WIFI_SSID       "IdoomFibre_ATbKR3yCR"
#define WIFI_PASSWORD   "malakikram5554"

#define SERVER_BASE_URL "https://edna-chasmic-unreasonably.ngrok-free.dev/api"
#define API_KEY         "esp32_shared_secret_key_here"
#define LOGIN_EMAIL     "ouledmeriemfarouk1@gmail.com"
#define LOGIN_PASSWORD  "artoriastm"
#define FIRMWARE_VER    "v1.0.0"

#define SEND_INTERVAL_MS  1000
#define WIFI_TIMEOUT_MS   15000
#define HTTP_TIMEOUT_MS   8000

// =============================================================================
//  INCLUDES
// =============================================================================
#include <Arduino.h>
#include <WiFi.h>
#include <WiFiClientSecure.h>
#include <HTTPClient.h>
#include <ArduinoJson.h>
#include <time.h>

// =============================================================================
//  STATE  — mirrors simulate_esp.js `state` object
// =============================================================================
String g_patientId = "";
bool   g_ntpSynced = false;
int    g_readingNum = 1;

float s_glucose     = 126.0f;
float s_heartRate   =  72.0f;
float s_temperature =  36.6f;
float s_oxygen      =  98.0f;
float s_systolic    = 120.0f;
float s_diastolic   =  80.0f;
float s_battery     =  85.0f;

// =============================================================================
//  HELPERS
// =============================================================================
float randAmp(float amplitude) {
  return ((float)random(0, 10001) / 10000.0f - 0.5f) * 2.0f * amplitude;
}

float clampf(float v, float lo, float hi) {
  return v < lo ? lo : (v > hi ? hi : v);
}

void drift() {
  s_glucose     = clampf(roundf(s_glucose     + randAmp(8.0f)),   60,  250);
  s_heartRate   = clampf(roundf(s_heartRate   + randAmp(5.0f)),   45,  150);
  s_temperature = clampf(roundf((s_temperature + randAmp(0.2f)) * 10) / 10.0f, 35.0f, 42.0f);
  s_oxygen      = clampf(roundf(s_oxygen      + randAmp(1.0f)),   88,  100);
  s_systolic    = clampf(roundf(s_systolic    + randAmp(5.0f)),   90,  200);
  s_diastolic   = clampf(roundf(s_diastolic   + randAmp(3.0f)),   50,  130);
  s_battery     = clampf(s_battery - (float)random(0, 200) / 10000.0f, 0, 100);
}

void buildTimestamp(char* buf, size_t len) {
  struct tm t;
  if (g_ntpSynced && getLocalTime(&t)) {
    strftime(buf, len, "%Y-%m-%dT%H:%M:%S.000Z", &t);
  } else {
    unsigned long s = millis() / 1000;
    snprintf(buf, len, "1970-01-01T%02lu:%02lu:%02lu.000Z",
             s / 3600, (s / 60) % 60, s % 60);
  }
}

// =============================================================================
//  WiFi
// =============================================================================
void connectWiFi() {
  Serial.printf("\n[WiFi] Connecting to \"%s\" …", WIFI_SSID);
  WiFi.mode(WIFI_STA);
  WiFi.begin(WIFI_SSID, WIFI_PASSWORD);
  unsigned long start = millis();
  while (WiFi.status() != WL_CONNECTED) {
    if (millis() - start > WIFI_TIMEOUT_MS) {
      Serial.println("\n[WiFi] TIMEOUT — restarting");
      delay(3000); ESP.restart();
    }
    delay(500); Serial.print(".");
  }
  Serial.printf("\n[WiFi] Connected! IP: %s\n", WiFi.localIP().toString().c_str());
}

void ensureWiFi() {
  if (WiFi.status() != WL_CONNECTED) {
    WiFi.disconnect();
    connectWiFi();
  }
}

// =============================================================================
//  NTP
// =============================================================================
void syncNTP() {
  Serial.println("[NTP] Syncing …");
  configTime(0, 0, "pool.ntp.org", "time.google.com");
  struct tm t; int tries = 0;
  while (!getLocalTime(&t) && tries++ < 20) { delay(500); Serial.print("."); }
  g_ntpSynced = (tries < 20);
  Serial.println(g_ntpSynced ? "\n[NTP] OK" : "\n[NTP] FAILED — using uptime stamps");
}

// =============================================================================
//  LOGIN — mirrors simulate_esp.js login()
// =============================================================================
bool serverLogin() {
  Serial.println("[Login] Authenticating …");

  WiFiClientSecure client; client.setInsecure();
  HTTPClient http;
  http.begin(client, String(SERVER_BASE_URL) + "/auth/login");
  http.addHeader("Content-Type", "application/json");
  http.setTimeout(HTTP_TIMEOUT_MS);

  String body = "{\"email\":\"" LOGIN_EMAIL "\",\"password\":\"" LOGIN_PASSWORD "\"}";
  int code = http.POST(body);
  Serial.printf("[Login] HTTP %d\n", code);

  if (code != 200 && code != 201) { http.end(); return false; }

  String payload = http.getString();
  http.end();

  Serial.println("[Login] Response: " + payload.substring(0, 200));

  StaticJsonDocument<4096> doc;
  if (deserializeJson(doc, payload)) { Serial.println("[Login] JSON parse failed"); return false; }

  const char* pid = doc["user"]["patient"]["id"];
  if (!pid) { Serial.println("[Login] patient.id missing"); return false; }

  g_patientId = String(pid);
  Serial.printf("[Login] patientId = %s\n", g_patientId.c_str());
  return true;
}

// =============================================================================
//  SEND — mirrors simulate_esp.js send()
// =============================================================================
void sendVitals() {
  drift();
  ensureWiFi();

  char ts[32];
  buildTimestamp(ts, sizeof(ts));

  bool fallDetected = (g_readingNum % 5 == 0);

  StaticJsonDocument<512> doc;
  doc["patientId"]    = g_patientId;
  doc["glucose"]      = (int)s_glucose;
  doc["heartRate"]    = (int)s_heartRate;
  doc["temperature"]  = s_temperature;
  doc["oxygen"]       = (int)s_oxygen;
  doc["systolic"]     = (int)s_systolic;
  doc["diastolic"]    = (int)s_diastolic;
  doc["batteryPct"]   = (int)s_battery;
  doc["firmwareVer"]  = FIRMWARE_VER;
  doc["recordedAt"]   = ts;
  doc["fallDetected"] = fallDetected;

  String body;
  serializeJson(doc, body);

  Serial.println("[Send] Payload: " + body);

  WiFiClientSecure client; client.setInsecure();
  HTTPClient http;
  http.begin(client, String(SERVER_BASE_URL) + "/vitals/ingest");
  http.addHeader("Content-Type", "application/json");
  http.addHeader("x-api-key", API_KEY);
  http.setTimeout(HTTP_TIMEOUT_MS);

  int code = http.POST(body);
  String resp = http.getString();
  http.end();

  Serial.printf("[Vitals] #%03d HTTP %d | %s\n", g_readingNum, code, resp.c_str());
  if (code == 201) Serial.println("[Vitals] OK");
  else if (code < 0) Serial.printf("[Vitals] Connection error: %s\n", HTTPClient::errorToString(code).c_str());

  g_readingNum++;
}

// =============================================================================
//  SETUP / LOOP
// =============================================================================
void setup() {
  Serial.begin(115200);
  delay(500);
  Serial.println("\n=== ESP32 Fake Vitals Test ===");
  randomSeed(analogRead(0));
  connectWiFi();
  syncNTP();
  while (!serverLogin()) {
    Serial.println("[Login] Retry in 5s …");
    delay(5000);
  }
  Serial.println("=== Starting send loop ===\n");
}

void loop() {
  unsigned long t = millis();
  sendVitals();
  unsigned long elapsed = millis() - t;
  if (elapsed < SEND_INTERVAL_MS) delay(SEND_INTERVAL_MS - elapsed);
}
